Future<EMCursorResults<EMMessage>> fetchHistoryMessages ({@required String conversationId, @required EMConversationType type, int pageSize: 10, String startMsgId, dynamic onSuccess(), dynamic onError(int code, String desc) })

fetchHistoryMessages - Fetches history messages in conversation conversationId, filtered by type. Result paginated by pageSize per page, started from startMsgId.

Implementation

Future<EMCursorResults<EMMessage>> fetchHistoryMessages(
    {@required final String conversationId,
    @required final EMConversationType type,
    int pageSize = 10,
    String startMsgId,
    onSuccess(),
    onError(int code, String desc)}) async {
  Map<String, dynamic> result = await _emChatManagerChannel
      .invokeMethod(EMSDKMethod.fetchHistoryMessages, {
    "id": conversationId,
    "type": toEMConversationType(type),
    "pageSize": pageSize,
    "startMsgId": startMsgId
  });
  if (result['success']) {
    return _EMCursorResults<EMMessage>(result['cursorId']);
  } else {
    return null;
  }
}
