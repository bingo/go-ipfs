Implementation

toEMDownloadStatus(EMDownloadStatus status){
  if(status == EMDownloadStatus.DOWNLOADING){
    return 0;
  } else if(status == EMDownloadStatus.SUCCESSED){
    return 1;
  } else if(status == EMDownloadStatus.FAILED){
    return 2;
  } else if(status == EMDownloadStatus.PENDING){
    return 3;
  }
}
